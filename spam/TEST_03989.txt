I think everybody has their own version of this,
but in case it's useful.. (only tested on Linux):

find $* \( -type f -o -type l \) -maxdepth 1 -mindepth 1 -print0 |
xargs -r0 du -b --max-depth 0 |
sort -k1n |
grep -v "^0"
